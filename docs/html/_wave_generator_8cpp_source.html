63 inline void WaveGenerator::switchPhase(VOICE &voice) {
64  switch (voice.phase) {
65  case VoicePhase::ATTACK:
66  if (voice.phaseOffset == voice.instrument->attackTime) {
67  voice.phaseOffset = 0;
68  voice.phase = VoicePhase::DECAY1;
69  }
70  break;
71  case VoicePhase::DECAY1:
72  if (voice.phaseOffset == voice.instrument->decay1Time) {
73  voice.phaseOffset = 0;
74  voice.phase = VoicePhase::SUSTAIN1;
75  }
76  break;
77  case VoicePhase::SUSTAIN1:
78  if (voice.phaseOffset == voice.instrument->sustain1Time) {
79  voice.phaseOffset = 0;
80  voice.phase = VoicePhase::DECAY2;
81  }
82  break;
83  case VoicePhase::DECAY2:
84  if (voice.phaseOffset == voice.instrument->decay1Time) {
85  voice.phaseOffset = 0;
86  voice.phase = VoicePhase::SUSTAIN2;
87  }
88  break;
89  case VoicePhase::RELEASE:
90  if (voice.phaseOffset == voice.instrument->releaseTime) {
91  voice.phase = VoicePhase::AVAILABLE;
92  }
93  break;
94  default:
95  break;
96  }
97 
98  if (voice.currentOffset == voice.duration) {
99  voice.phaseOffset = 0;
100  voice.phase = VoicePhase::RELEASE;
101  }
102 }

110 inline AMPLITUDE WaveGenerator::currentAmplitude(VOICE &voice) {
111  uint32_t amplitude = voice.velocity;
112 
113  switch (voice.phase) {
114  case VoicePhase::ATTACK:
115  amplitude *= voice.phaseOffset;
116  amplitude /= voice.instrument->attackTime;
117  voice.amplitude = amplitude;
118  break;
119  case VoicePhase::DECAY1:
120  amplitude -= voice.phaseOffset * (voice.velocity - voice.instrument->decay1Amount) / voice.instrument->decay1Time;
121  voice.amplitude = amplitude;
122  break;
123  case VoicePhase::SUSTAIN1:
124  amplitude *= voice.instrument->decay1Amount;
125  voice.amplitude = amplitude;
126  break;
127  case VoicePhase::DECAY2:
128  amplitude -= voice.phaseOffset * (voice.velocity - voice.instrument->decay2Amount) / voice.instrument->decay2Time;
129  voice.amplitude = amplitude;
130  break;
131  case VoicePhase::SUSTAIN2:
132  amplitude *= voice.instrument->decay2Amount;
133  voice.amplitude = amplitude;
134  break;
135  case VoicePhase::RELEASE:
136  amplitude = voice.amplitude * (voice.instrument->releaseTime - voice.phaseOffset) / voice.instrument->releaseTime;
137  break;
138  default:
139  break;
140  }
141 
142  return amplitude;
143 }

150 inline SAMPLE WaveGenerator::renderVoice(VOICE &voice) {
151  if (voice.phase == VoicePhase::AVAILABLE) {
152  return 0;
153  }
154 
155  uint8_t sampleOffset = uint32_t(voice.currentOffset * voice.frequency / 440) % 72;
156  SAMPLE sample = pgm_read_byte(voice.instrument->samples + sampleOffset);
157 
158 // Calculate amplitude
159  AMPLITUDE amplitude = currentAmplitude(voice);
160 
161  int16_t bigSample = sample * amplitude;
162  bigSample /= _voiceCount;
163  bigSample /= 256;
164 
165 
166  voice.currentOffset++;
167  voice.phaseOffset++;
168 
169  switchPhase(voice);
170 
171  return (SAMPLE)bigSample;
172 }
