64  void pushBits(uint8_t byte, bool rs) {
65  // setup RS
66  byte |= rs;
67 
68  // Load the byte into the shift register
69  _shifter.shiftOut(byte);

71  // strobe Enable pin

84  void writeByte(uint8_t byte, bool rs) {
85  if (Display_HD44780<cols, rows, txBuffers>::_byteMode) {
86  // 8 bits interface mode
87  pushBits(byte, rs);
88  } else {
89  // 4 bits interface mode
90  // Write high order nibble
91  pushBits(byte & 0xF0, rs);
92  // Write low order nibble
93  pushBits(byte << 4, rs);
