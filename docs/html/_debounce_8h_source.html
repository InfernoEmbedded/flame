90  void pinChanged(uint8_t pcInt, bool newState) {
91  if (newState) {
92  // Pin has been released

94 
95  /* Call the singlePress function if the time is greater than the debounce time,
96  * and the button has not been held down
97  */
98  _rtc.elapsed(_pins[pcInt].timestamp, heldFor);
99  uint16_t heldMilliseconds = getMilliseconds(heldFor);
100  if (!_pins[pcInt].held && heldMilliseconds >= debounceTime) {
