380  uint8_t calculatePrescaler(uint32_t time, TIMER_PRESCALER *prescaler, uint16_t *factor) {
381  uint32_t limit = 0;
382  if (8 == bits) {
383  limit = 256L;
384  } else if (16 == bits) {
385  limit = 65536L;
386  }
387 
388  switch (type) {
389  case TIMER_TYPE::HAS_5_PRESCALERS:
390  if (time <= limit) {
391  *prescaler = TIMER_PRESCALER::PRESCALER_5_1;
392  *factor = 1;
393  } else if (time <= limit * 8) {
394  *prescaler = TIMER_PRESCALER::PRESCALER_5_8;
395  *factor = 8;
396  } else if (time <= limit * 64) {
397  *prescaler = TIMER_PRESCALER::PRESCALER_5_64;
398  *factor = 64;
399  } else if (time <= limit * 256) {
400  *prescaler = TIMER_PRESCALER::PRESCALER_5_256;
401  *factor = 256;
402  } else if (time <= limit * 1024) {
403  *prescaler = TIMER_PRESCALER::PRESCALER_5_1024;
404  *factor = 1024;
405  } else {
406  return 1;
407  }
408  break;
409 
410  case TIMER_TYPE::HAS_7_PRESCALERS:
411  if (time <= limit) {
412  *prescaler = TIMER_PRESCALER::PRESCALER_7_1;
413  *factor = 1;
414  } else if (time <= limit * 8) {
415  *prescaler = TIMER_PRESCALER::PRESCALER_7_8;
416  *factor = 8;
417  } else if (time <= limit * 32) {
418  *prescaler = TIMER_PRESCALER::PRESCALER_7_32;
419  *factor = 32;
420  } else if (time <= limit * 64) {
421  *prescaler = TIMER_PRESCALER::PRESCALER_7_64;
422  *factor = 64;
423  } else if (time <= limit * 128) {
424  *prescaler = TIMER_PRESCALER::PRESCALER_7_128;
425  *factor = 128;
426  } else if (time <= limit * 256) {
427  *prescaler = TIMER_PRESCALER::PRESCALER_7_256;
428  *factor = 256;
429  } else if (time <= limit * 1024) {
430  *prescaler = TIMER_PRESCALER::PRESCALER_7_1024;
431  *factor = 1024;
432  } else {
433  return 1;
434  }
435  break;
436  }
437 
438  return 0;
439  }
