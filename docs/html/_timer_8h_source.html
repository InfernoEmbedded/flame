497  uint8_t calculatePrescaler(uint32_t time, TimerPrescaler *prescaler, uint16_t *factor) {
498  uint32_t limit = 0;
499  if (8 == bits) {
500  limit = 256L;
501  } else if (16 == bits) {
502  limit = 65536L;
503  }
504 
505  switch (type) {
506  case TimerType::HAS_5_PRESCALERS:
507  if (time <= limit) {
508  *prescaler = TimerPrescaler::PRESCALER_5_1;
509  *factor = 1;
510  } else if (time <= limit * 8) {
511  *prescaler = TimerPrescaler::PRESCALER_5_8;
512  *factor = 8;
513  } else if (time <= limit * 64) {
514  *prescaler = TimerPrescaler::PRESCALER_5_64;
515  *factor = 64;
516  } else if (time <= limit * 256) {
517  *prescaler = TimerPrescaler::PRESCALER_5_256;
518  *factor = 256;
519  } else if (time <= limit * 1024) {
520  *prescaler = TimerPrescaler::PRESCALER_5_1024;
521  *factor = 1024;
522  } else {
523  return 1;
524  }
525  break;
526 
527  case TimerType::HAS_7_PRESCALERS:
528  if (time <= limit) {
529  *prescaler = TimerPrescaler::PRESCALER_7_1;
530  *factor = 1;
531  } else if (time <= limit * 8) {
532  *prescaler = TimerPrescaler::PRESCALER_7_8;
533  *factor = 8;
534  } else if (time <= limit * 32) {
535  *prescaler = TimerPrescaler::PRESCALER_7_32;
536  *factor = 32;
537  } else if (time <= limit * 64) {
538  *prescaler = TimerPrescaler::PRESCALER_7_64;
539  *factor = 64;
540  } else if (time <= limit * 128) {
541  *prescaler = TimerPrescaler::PRESCALER_7_128;
542  *factor = 128;
543  } else if (time <= limit * 256) {
544  *prescaler = TimerPrescaler::PRESCALER_7_256;
545  *factor = 256;
546  } else if (time <= limit * 1024) {
547  *prescaler = TimerPrescaler::PRESCALER_7_1024;
548  *factor = 1024;
549  } else {
550  return 1;
551  }
552  break;
553  }
554 
555  return 0;
556  }
