206  void init(bool byteMode, bool multiLine, bool bigFont, bool cursorOn, bool cursorBlink,
207  bool left2right, bool scroll) {
208  /* Assume the worst case, we are called immediately after poweron
209  * Also assume that the MCU init takes 0 time
210  */
211  _delay_ms(HD44780_TINIT);
212 
213  // hardware initialization always set 8 bits mode
214  _byteMode = true;
215  uint8_t resetData = 1 << 4 | multiLine << 3 | bigFont << 2 | 1;
216  writeCommand(HD44780_COMMAND::SET_FUNCTION, resetData);
217  writeCommand(HD44780_COMMAND::SET_FUNCTION, resetData);
218  writeCommand(HD44780_COMMAND::SET_FUNCTION, resetData);
219 
220  _byteMode = byteMode;
221 
222 
223  function(byteMode, multiLine, bigFont);
224  _delay_us(39);
225  control(true, cursorOn, cursorBlink);
226  clear();
227  entryMode(left2right, scroll);
228  Display_Character<cols, rows, txBuffers>::_currentRow = rows - 1;
229  Display_Character<cols, rows, txBuffers>::_currentCol = 0;
230  }
