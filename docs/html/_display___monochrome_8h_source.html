65  bool writeChar(const FONT &font, int16_t *offsetX, int16_t offsetY,
66  uint8_t onValue, uint8_t offValue, char character) {
67  uint8_t c = (uint8_t)character;
68  if (c < font.firstChar || c >= font.firstChar + font.charCount) {
69  c = font.unknown;
70  }
71 
72  // Normalize character to font
73  c -= font.firstChar;
74 #ifdef __FLASH
75  uint8_t charWidth = font.widths[c];
76 
77  const uint8_t *fontChar = font.fontData + font.offsets[c];
78 #else
79  uint8_t charWidth = pgm_read_byte(font.widths + c);
80 
81  const uint8_t *fontChar = font.fontData + pgm_read_word(font.offsets + c);
82 #endif
83 
84  // Render each column
85  bool ret = false;
86  do {
87  int8_t bit = 7;
88  uint8_t y = 0;
89 #ifdef __FLASH
90  uint8_t data = *(fontChar++);
91 #else
92  uint8_t data = pgm_read_byte(fontChar++);
93 #endif
94 
95  if (*offsetX >= 0) {
96  // Start at the bottom of the column and work up
97  for (; y < font.maxHeight; y++, bit--) {
98  ret = true;
99 
100  if (data & (1 << bit)) {
101  setPixel(*offsetX, offsetY + y, onValue);
102  } else {
103  setPixel(*offsetX, offsetY + y, offValue);
104  }
105 
106  // Support multiple-byte font images - fetch the next byte of data if we exhaust the current one
107  if (-1 == bit && (font.maxHeight - y) > 0) {
108  bit = 7;
109 #ifdef __FLASH
110  data = *(fontChar++);
111 #else
112  data = pgm_read_byte(fontChar++);
113 #endif
114  }
115  }
116  }
117  } while (++(*offsetX) < (int16_t)cols && --charWidth);
118 
119  *offsetX += charWidth;
120 
121  return ret;
122  }

132  bool writeSeperator(const FONT &font, int16_t *offsetX, int16_t offsetY,
133  uint8_t offValue) {
134  bool ret = false;
135 
136  if (*offsetX >= 0 && *offsetX < (int16_t)cols) {
137  // Start at the bottom of the column and work up
138  uint8_t y = 0;
139  uint8_t row;
140 
141  while (y++ < font.maxHeight) {
142  row = offsetY + y;
143  if (row < rows) {
144  setPixel(*offsetX, row, offValue);
145  ret = true;
146  }
147  }
148  }
149

201  bool writeBuffer(const FONT &font, int16_t *offsetX, int16_t offsetY,
202  uint8_t onValue, uint8_t offValue, TXBuffer &txBuffer) {
203  bool ret = false;
204  while (*offsetX < (int16_t)cols) {
205  int c = txBuffer.nextCharacter();

207  break;
208  }
209

211 
212  if (txBuffer.hasMore()) {

214  }
215  }
