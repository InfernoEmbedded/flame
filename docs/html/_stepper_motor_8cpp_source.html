94 void StepperMotor::alarm() {
95  if (!_moving) {
96  return;
97  }
98 
99  TIMESTAMP elapsed;
100  _rtc.elapsed(_started, elapsed);
101 // Reuse elapsed.timestamp to represent milliseconds elapsed
102  elapsed.timestamp *= 1000;
103  elapsed.timestamp += elapsed.milliseconds;
104 
105  int32_t expectedPosition = _startPosition + elapsed.timestamp * _speed / ((_forward) ? 1000 : -1000);
106 
107  if (expectedPosition != _position) {
108  step(_forward);
109  _position += (_forward) ? 1 : -1;
110  }
111 
112  if (_until == _position) {
113 // We are done
114  _speed = 0;
115  _moving = false;
116  _rtc.removeAlarm(this);
117  if (_stepperListener) {
118  _stepperListener->moveComplete(_position);
119  }
120  }
121 }
