226 void TripleAxisSensor::handleEvents() {
227  if (isSampleReady()) {
228  ATOMIC_BLOCK(ATOMIC_RESTORESTATE) {
229  memcpy((void *)&_valueOut, (void *)&_valueTemp, sizeof(_valueOut));
230  }
231 
232  if (_limitMagnitudeSquared) {
233  int32_t magSquared = magnitudeSquared();
234 
235  if (magSquared >= _limitMagnitudeSquared) {
236  _listener->limitReached(*this, TripleAxisSensorChannel::MAGNITUDE);
237  }
238  }
239 
240  if (_listener) {
241  TRIPLEAXISSENSOR_READING value;
242  getValue(&value);
243 
244  for (uint8_t i = 0; i < 3; i++) {
245  if (_limit.value[i] && (value.value[i] * value.value[i]) >= _limit.value[i]) {
246  _listener->limitReached(*this, (TripleAxisSensorChannel)i);
247  }
248  }
249 
250  _listener->sampleIsReady(*this);
251  }
252  }
253 }
