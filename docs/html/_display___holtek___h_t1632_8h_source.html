302

326  void setPixel(uint16_t col, uint16_t row, uint8_t value) {
327  if (row < DISPLAY_Y && col < DISPLAY_X) {
328  // Coordinates of the display module
329  uint8_t moduleX, moduleY;
330 
331  moduleX = col / MODULE_X;
332  moduleY = row / MODULE_Y;
333 
334  // offsets within the module
335  col -= moduleX * MODULE_X;
336  row -= moduleY * MODULE_Y;
337 
338  uint8_t offset = col;
339  uint8_t bit = row;
340  switch (mode) {
341  case HT1632_MODE::NMOS_32x8:
342  case HT1632_MODE::PMOS_32x8:
343  break;
344  case HT1632_MODE::NMOS_24x16:
345  case HT1632_MODE::PMOS_24x16:
346  offset *= 2;
347  if (bit > 7) {
348  bit -= 8;
349  offset++;
350  }
351  break;
352  }
353 
354  // Change offset from the start of the display to the start of the framebuffer
355  offset += (moduleY * arrayX + moduleX) * DISPLAY_BYTES;
356 
357  if (value) {
358  _frameBuffer[offset] |= 1 << bit;
359  } else {
360  _frameBuffer[offset] &= ~(1 << bit);
361  }
362  }
363  }
