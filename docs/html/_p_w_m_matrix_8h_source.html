78  inline void tickRow(void) {
79  uint16_t i;
80 
81  if (0 == _currentLevel) {
82  // Turn on the current row
83  _driver.rowOn(_currentRow);
84  for (i = 0; i < cols; i++) {
85  if (pixel(i, _currentRow) > 0) {
86  _driver.colOn(i);
87  }
88  }
89  } else {
90  // Turn off pixels that get switched off on this pass
91  for (i = 0; i < cols; i++) {
92  if (pixel(i, _currentRow) <= _currentLevel) {
93  _driver.colOff(i);
94  }
95  }
96  }
97 
98  if (255 == ++_currentLevel) {
99  // Turn off the current row & advance
100  _driver.rowOff(_currentRow);
101 
102  _currentLevel = 0;
103  if (++_currentRow == rows) {
104  _currentRow = 0;
105  }
106  }
107  }

146  inline void tickPixel(void) {
147  if (0 == _currentLevel) {
148  // Turn on the pixel at the current row & column
149  _driver.colOn(_currentCol);
150  _driver.rowOn(_currentRow);
151  } else if (pixel(_currentRow, _currentCol) <= _currentLevel) {
152  // Turn off the pixel
153  _driver.colOff(_currentCol);
154  _driver.rowOff(_currentRow);
155  }
156 
157  if (255 == ++_currentLevel) {
158  // Advance the column
159  if (++_currentCol == cols) {
160  _currentCol = 0;
161 
162  // Advance the row
163  if (++_currentRow == rows) {
164  _currentRow = 0;
165  }
166  }
167  }
168  }
