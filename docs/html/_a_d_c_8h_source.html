124  uint16_t MHV_ADC::busyRead(int8_t channel, uint8_t reference) {
125  ADMUX = reference | (channel & 0x0F);
126 
127 #ifdef MUX5
128  ADCSRB = (ADCSRB & ~_BV(MUX5)) | ((channel & _BV(5) >> (5 - MUX5)));
129 #endif
130 
131  // Start the conversion
132  ADCSRA |= _BV(ADSC);
133 
134  while (ADCSRA & _BV(ADSC)) {
135  };
136 
137  return ADC;
