77  int asyncReadLine(char *buffer, uint8_t bufferLength) {
78  bool isFull = _rxBuffer.full();
79  if (!isFull) {
80  // Peek at the last character & see if its a newline
81  char last = (char)_rxBuffer.peekHead();
82 
83  if ('\r' != last && '\n' != last) {
84  return -1;
85  }
86  }
87 
88  uint8_t i = 0;
89  int byte;
90  while (-1 != (byte = _rxBuffer.consume())) {
91  if (i == bufferLength - 1) {
92  buffer[i] = '\0';
93  return -2;
94  }
95  if ('\r' == byte || '\n' == byte) {
96  break;
97  }
98  buffer[i++] = (char)byte;
99  }

104  }
