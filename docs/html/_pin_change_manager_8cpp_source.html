70 void PinChangeManager::pinChange(uint8_t offset) {
71  EVENT_PIN *pin = _pins + offset;
72  EVENT_PIN *maxPin = pin + 8;
73 
74  for (; pin < maxPin; ++pin) {
75  if (NULL == pin->listener) {
76  continue;
77  }
78 
79  bool cur = _MMIO_BYTE(pin->port) & pin->mask;
80  if (cur != pin->previous) {
81  // Pin has changed
82  pin->previous = cur;
83  pin->changed = 1;
84  pin->listener->pinChanged(pin - _pins, cur);
85  }
86  }
87 }
