67 /* Calibrate the RC oscillator to 8.25 MHz. The core clock of 16.5 MHz is
68  * derived from the 66 MHz peripheral clock by dividing. Our timing reference
69  * is the Start Of Frame signal (a single SE0 bit) available immediately after
70  * a USB RESET. We first do a binary search for the OSCCAL value and then
71  * optimize this value with a neighboorhod search.
72  * This algorithm may also be used to calibrate the RC oscillator directly to
73  * 12 MHz (no PLL involved, can therefore be used on almost ALL AVRs), but this
74  * is wide outside the spec for the OSCCAL value and the required precision for
75  * the 12 MHz clock! Use the RC oscillator calibrated to 12 MHz for
76  * experimental purposes only!
77  */
78 static void calibrateOscillator(void)
79 {
80 uchar step = 128;
81 uchar trialValue = 0, optimumValue;
82 int x, optimumDev, targetValue = (unsigned)(1499 * (double)F_CPU / 10.5e6 + 0.5);
83 
84  /* do a binary search: */
85  do{
86  OSCCAL = trialValue + step;
87  x = usbMeasureFrameLength(); /* proportional to current real frequency */
88  if(x < targetValue) /* frequency still too low */
89  trialValue += step;
90  step >>= 1;
91  }while(step > 0);
92  /* We have a precision of +/- 1 for optimum OSCCAL here */
93  /* now do a neighborhood search for optimum value */
94  optimumValue = trialValue;
95  optimumDev = x; /* this is certainly far away from optimum */
96  for(OSCCAL = trialValue - 1; OSCCAL <= trialValue + 1; OSCCAL++){
97  x = usbMeasureFrameLength() - targetValue;
98  if(x < 0)
99  x = -x;
100  if(x < optimumDev){
101  optimumDev = x;
102  optimumValue = OSCCAL;
103  }
104  }
105  OSCCAL = optimumValue;
106 }
107 /*
108 Note: This calibration algorithm may try OSCCAL values of up to 192 even if
109 the optimum value is far below 192. It may therefore exceed the allowed clock
110 frequency of the CPU in low voltage designs!
111 You may replace this search algorithm with any other algorithm you like if
112 you have additional constraints such as a maximum CPU clock.
113 For version 5.x RC oscillators (those with a split range of 2x128 steps, e.g.
114 ATTiny25, ATTiny45, ATTiny85), it may be useful to search for the optimum in
115 both regions.
116 */

129 static unsigned char mhv_vusbIdleRate = 1; /* in 4 ms units */

131 unsigned char usbFunctionSetup(uchar data[8]) {
132  usbRequest_t *rq = (usbRequest_t *) ((void *) data);
133 
134  usbMsgPtr = mhv_vusbReportBuffer;
135  if ((rq->bmRequestType & USBRQ_TYPE_MASK) == USBRQ_TYPE_CLASS) { /* class request type */
136  if (rq->bRequest == USBRQ_HID_GET_REPORT) { /* wValue: ReportType (highbyte), ReportID (lowbyte) */
137  /* we only have one report type, so don't look at wValue */
138  return sizeof(mhv_vusbReportBuffer);
139  } else if (rq->bRequest == USBRQ_HID_GET_IDLE) {
140  usbMsgPtr = &mhv_vusbIdleRate;
141  return 1;
142  } else if (rq->bRequest == USBRQ_HID_SET_IDLE) {
143  mhv_vusbIdleRate = rq->wValue.bytes[1];
144  }
145  } else {
146  /* no vendor specific requests implemented */
147  }
148  return 0;
149 }
