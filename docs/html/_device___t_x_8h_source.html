669  file, line, function);
670  length += vsnprintf_P(NULL, 0, format, ap);
671  length += 3; // "\r\n\0"
672 
673  char *buf;
674  ATOMIC_BLOCK(ATOMIC_RESTORESTATE) {
675  buf = (char *)malloc(length);
676  }
677  char *cur = buf;
678  if (NULL != cur) {
679  cur += snprintf_P(cur, length, prefix,
680  file, line, function);

682  cur += vsnprintf_P(cur, lengthRemaining, format, ap);
683  *(cur++) = '\r';
684  *(cur++) = '\n';
