240  char c = '\0';
241 
242  if (!_isString && offset >= (_length - 1)) {
243  return -1;

246  switch (_type) {

261  case MEMX:
262  const __memx char *memx = _data + offset;
263  c = *memx;
264  break;
265 #endif
266 #endif
267  }

269  if ((_isString && '\0' == c)) {
270  return -1;

619  PGM_P format, ...) {
620  va_list ap;
621  va_start(ap, format);
622 
623  int length = snprintf_P(NULL, 0, PSTR("%s:%d\t%s():\t\t"),
624  file, line, function);
625  length += vsnprintf_P(NULL, 0, format, ap);
626  length += 3; // "\r\n\0"
627 
628  char *buf = (char *)malloc(length);
629  char *cur = buf;
630  if (NULL != cur) {
631  cur += snprintf_P(cur, length, PSTR("%s:%d\t%s():\t\t"),
632  file, line, function);
633  length -= cur - buf;
634  cur += vsnprintf_P(cur, length, format, ap);
635  *(cur++) = '\r';
636  *(cur++) = '\n';
637  *cur = '\0';

640  device_tx_free(buf);
641  va_end(ap);
