191  void configure(SerialMode mode, uint8_t bits, SerialParity parity, uint8_t stopBits, uint8_t polarity) {
192  stopBits -= 1;
193 
194  uint8_t frameSizeMask;
195 
196  switch (bits) {
197  case 5:
198  frameSizeMask = 0;
199  break;
200  case 6:
201  frameSizeMask = 1;
202  break;
203  case 7:
204  frameSizeMask = 2;
205  break;
206  case 8:
207  default:
208  frameSizeMask = 3;
209  break;
210 
211  case 9:
212  frameSizeMask = 3;
213  _MMIO_BYTE(usartControlB) &= 1 << 2;
214  break;
215 
216  }
217 
218  _MMIO_BYTE(usartControlC) = (mode << 6) | (parity << 4) | (frameSizeMask << 1) | polarity;
219  }

276  INLINE void setSpeed(unsigned long newBaud) {
277  /* Use U2X if the requested baud rate is higher than (F_CPU/16),
278  * otherwise use whatever has the least error
279  */
280  if (newBaud > F_CPU / 16 ||
281  abs((int)(255-((F_CPU/(8*(((F_CPU/4/newBaud-1)/2)+1))*255)/newBaud))) < abs((int)(255-((F_CPU/(16*(((F_CPU/8/newBaud-1)/2)+1))*255)/newBaud)))) {
282  _MMIO_BYTE(usartStatus) = _BV(usartU2X);
283  _MMIO_BYTE(usartBaud) = (uint16_t)(F_CPU / 4 / newBaud - 1) / 2;
284  } else {
285  _MMIO_BYTE(usartStatus) = 0;
286  _MMIO_BYTE(usartBaud) = (uint16_t)(F_CPU / 8 / newBaud - 1) / 2;
287  }
288 
289  // Enable the port
290  _MMIO_BYTE(usartControlB) |= _BV(usartRxEnable) | _BV(usartTxEnable) | _BV(usartRxInterruptEnable);
291  }
