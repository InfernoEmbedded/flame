76 float PID::compute(float input) {
77  float error = _setpoint - input;
78 
79  _integral += (_kI * error);
80 
81  clampIntegral();
82 
83  float output = _kP * error + _integral - _kD * (input - _lastInput);
84 
85  if (output > _outMax)
86  output = _outMax;
87  else if (output < _outMin)
88  output = _outMin;
89 
90  _lastInput = input;
91  _lastOutput = output;
92 
93  return output;
94 }
