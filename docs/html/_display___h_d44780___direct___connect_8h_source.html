110  void writeNibble(uint8_t nibble, bool rs) {
111  uint8_t oldPort = _SFR_IO8(dataOut) & ~(_dataMask);
112  _SFR_IO8(controlOut) &= ~(HD44780_RW);
113  if (rs) {
114  _SFR_IO8(controlOut) |= HD44780_RS;
115  }
116 
117  _SFR_IO8(dataOut) = oldPort | ((nibble << dataPin) & _dataMask);
118 
119  _delay_loop_1(HD44780_TSU1);
120  _SFR_IO8(controlOut) |= HD44780_E;
121  _delay_loop_1(HD44780_TW);
122  _SFR_IO8(controlOut) &= ~HD44780_E & ~HD44780_RS;
123  _delay_loop_1(HD44780_TW);
124  }
