151  uint8_t fade(uint8_t initial, uint8_t final, int16_t iteration, int16_t total) {
152  if (0 == total || iteration >= total) {
153  return final;
154  }
155 
156  if (iteration <= 0) {
157  return initial;
158  }
159 
160  int32_t di = iteration;
161  di *= final - initial;
162  di /= total;
163 
164  return initial + di;
165  }
