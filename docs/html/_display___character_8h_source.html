83  uint16_t currentPosition = txBuffer.getPosition();
84

416  void writeChar(char character) {
417  switch (character) {
418  case '\b':
419  if (_wrap && 0 == _currentCol) {
420  // Go to the previous line
421  if (_currentRow + 1 < rows) {
422  _currentRow++;
423  }
424  setCursor(cols - 1, _currentRow);
425  } else {
426  setCursor(_currentCol - 1, _currentRow);
427  }
428  break;
429  case '\t':
430  _currentCol = (_currentCol + 4) & ~(4 - 1);
431  if (_currentCol >= cols) {
432  if (_wrap) {
433  if (_currentRow > 0) {
434  _currentRow--;
435  } else if (_scroll) {
436  scrollVertically();
437  break;
438  }
439  _currentCol = 0;
440  } else {
441  _currentCol = cols - 1;
442  }
443  }
444  setCursor(_currentCol, _currentRow);
445  break;
446  case '\n':
447  if (_currentRow > 0) {
448  _currentRow--;
449  } else if (_scroll) {
450  scrollVertically();
451  }
452  _currentCol = 0;
453  setCursor(_currentCol, _currentRow);
454  break;
455  default:
456  if (_currentCol++ >= cols - 1 && _wrap) {
457  _currentCol = 0;
458  if (_currentRow > 0) {
459  _currentRow--;
460  } else if (_scroll) {
461  scrollVertically();
462  } else {
463  setCursor(_currentCol, _currentRow);
464  }
465  }
466  _writeChar(character);
467  break;
468  }
469  }

474  void scrollVertically() {
475  uint16_t x, y;
476  char c;
477 
478  for (y = rows - 2; y > 0; y--) {
479  // Copy each character in this row to the row above
480  for (x = 0; x < cols; ++x) {
481  _setCursor(x, y + 1);
482  c = _readChar();
483  _setCursor(x, y);
484  _writeChar(c);
485  }
486  }
487 
488  // Blank the last row
489  _setCursor(0, 0);
490  for (x = 0; x < cols; ++x) {
491  _writeChar(' ');
492  }
493 
494  // Set the cursor ready to write another line
495  setCursor(0, 0);
496  }
