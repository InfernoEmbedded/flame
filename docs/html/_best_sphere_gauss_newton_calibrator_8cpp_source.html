67 void BestSphereGaussNewtonCalibrator::addObservation(const TRIPLEAXISSENSOR_RAW_READING &sample) {
68  _observationCount++;
69 
70  int32_t squareObs[AXIS_COUNT];
71  for (uint8_t i = 0; i < AXIS_COUNT; i++) {
72  squareObs[i] = static_cast<int32_t>(sample.value[i]) * sample.value[i];
73  }
74 
75  for (uint8_t i = 0; i < AXIS_COUNT; ++i) {
76  //Keep track of min and max in each dimension
77  _obsMin[i] = (sample.value[i] < _obsMin[i]) ? sample.value[i] : _obsMin[i];
78  _obsMax[i] = (sample.value[i] > _obsMax[i]) ? sample.value[i] : _obsMax[i];
79 
80  //accumulate sum and sum of squares in each dimension
81  _mu[i] += sample.value[i];
82  _mu2[i] += squareObs[i];
83 
84  //accumulate inner products of the vector of observations and the vector of squared observations.
85  for (uint8_t j = 0; j < AXIS_COUNT; ++j) {
86  _ipX2X[i][j] += squareObs[i] * static_cast<int32_t>(sample.value[j]);
87  if (i <= j) {
88  uint8_t idx = upperTriangularIndex(i, j);
89  _ipXX[idx] += static_cast<int32_t>(sample.value[i]) * static_cast<int32_t>(sample.value[j]);
90  _ipX2X2[idx] += squareObs[i] * static_cast<float>(squareObs[j]);
91  }
92  }
93  }

130 void BestSphereGaussNewtonCalibrator::computeGNMatrices(float JtJ[][6], float JtR[]) {
131  float beta2[6]; //precompute the squares of the model parameters
132  for (uint8_t i = 0; i < 6; ++i) {
133  beta2[i] = pow(_beta[i], 2);
134  }
135 
136 //compute the inner product of the vector of residuals with the constant 1 vector, the vector of
137 // observations, and the vector of squared observations.
138  float r = _observationCount; //sum of residuals
139  float rx[3]; //Inner product of vector of residuals with each observation vector
140  float rx2[3]; //Inner product of vector of residuals with each square observation vector
141 
142 //now correct the r statistics
143  for (uint8_t i = 0; i < 3; ++i) {
144  r -= (beta2[i] * _observationCount + _mu2[i] - 2 * _beta[i] * _mu[i]) / beta2[3 + i];
145  rx[i] = _mu[i];
146  rx2[i] = _mu2[i];
147  for (uint8_t j = 0; j < 3; ++j) {
148  rx[i] -= (beta2[j] * _mu[i] + _ipX2X[j][i] - 2 * _ipXX[upperTriangularIndex(i, j)] * _beta[j])
149  / beta2[3 + j];
150  rx2[i] -= (beta2[j] * _mu2[i] + _ipX2X2[upperTriangularIndex(i, j)] - 2 * _ipX2X[i][j] * _beta[j])
151  / beta2[3 + j];
152  }
153  }
154 
155  for (uint8_t i = 0; i < 3; ++i) {
156  //Compute product of Jacobian matrix with the residual vector
157  JtR[i] = 2 * (rx[i] - _beta[i] * r) / beta2[3 + i];
158  JtR[3 + i] = 2 * (rx2[i] - 2 * _beta[i] * rx[i] + beta2[i] * r) / (beta2[3 + i] * _beta[3 + i]);
159 
160  //Now compute the product of the transpose of the jacobian with itself
161  //Start with the diagonal blocks
162  for (uint8_t j = i; j < 3; ++j) {
163  JtJ[i][j] = JtJ[j][i] = 4
164  * (_ipXX[upperTriangularIndex(i, j)] - _beta[i] * _mu[j] - _beta[j] * _mu[i]
165  + _beta[i] * _beta[j] * _observationCount) / (beta2[3 + i] * beta2[3 + j]);
166  JtJ[3 + i][3 + j] = JtJ[3 + j][3 + i] = 4
167  * (_ipX2X2[upperTriangularIndex(i, j)] - 2 * _beta[j] * _ipX2X[i][j] + beta2[j] * _mu2[i]
168  - 2 * _beta[i] * _ipX2X[j][i] + 4 * _beta[i] * _beta[j] * _ipXX[upperTriangularIndex(i, j)]
169  - 2 * _beta[i] * beta2[j] * _mu[i] + beta2[i] * _mu2[j] - 2 * beta2[i] * _beta[j] * _mu[j]
170  + beta2[i] * beta2[j] * _observationCount) / pow(_beta[3 + i] * _beta[3 + j], 3);
171  }
172  //then get the off diagonal blocks
173  for (uint8_t j = 0; j < 3; ++j) {
174  JtJ[i][3 + j] = JtJ[3 + j][i] = 4
175  * (_ipX2X[j][i] - 2 * _beta[j] * _ipXX[upperTriangularIndex(i, j)] + beta2[j] * _mu[i]
176  - _beta[i] * _mu2[j] + 2 * _beta[i] * _beta[j] * _mu[j] - _beta[i] * beta2[j] * _observationCount)
177  / (beta2[3 + i] * beta2[3 + j] * _beta[3 + j]);
178  }
179  }

182 void BestSphereGaussNewtonCalibrator::findDelta(float JtJ[][6], float JtR[]) {
183 //Solve 6-d matrix equation JtJS*x = JtR
184 //first put in upper triangular form
185 //Serial.println("find delta");
186  float lambda;
187 
188 //make upper triangular
189  for (uint8_t i = 0; i < 6; ++i) {
190  //eliminate all nonzero entries below JS[i][i]
191 
192  for (uint8_t j = i + 1; j < 6; ++j) {
193  lambda = JtJ[j][i] / JtJ[i][i];
194  if (lambda != 0.0) {
195  JtR[j] -= lambda * JtR[i];
196  for (uint8_t k = i; k < 6; ++k) {
197  JtJ[j][k] -= lambda * JtJ[i][k];
198  }
199  }
200  }
201  }
202 
203 //back-substitute
204  for (int8_t i = 5; i >= 0; --i) {
205  JtR[i] /= JtJ[i][i];
206  JtJ[i][i] = 1.0;
207  for (int8_t j = 0; j < i; ++j) {
208  lambda = JtJ[j][i];
209  JtR[j] -= lambda * JtR[i];
210  JtJ[j][i] = 0.0;
211  }
212  }

215 void BestSphereGaussNewtonCalibrator::calculate() {
216  guessParameters();
217  float JtJ[6][6];
218  float JtR[6];
219  clearGNMatrices(JtJ, JtR);
220 
221  float eps = 0.000000001;
222  int num_iterations = 20;
223  float change = 100.0;
224  while (--num_iterations >= 0 && change > eps) {
225 
226  computeGNMatrices(JtJ, JtR);
227  findDelta(JtJ, JtR);
228 
229  change = JtR[0] * JtR[0] + JtR[1] * JtR[1] + JtR[2] * JtR[2] + JtR[3] * JtR[3] * (_beta[3] * _beta[3])
230  + JtR[4] * JtR[4] * (_beta[4] * _beta[4]) + JtR[5] * JtR[5] * (_beta[5] * _beta[5]);
231 
232  if (!isnan(change)) {
233  for (uint8_t i = 0; i < 6; ++i) {
234  _beta[i] -= JtR[i];
235 
236  if (i >= 3) {
237  _beta[i] = fabs(_beta[i]);
238  }
239  }
240  }
241 
242  clearGNMatrices(JtJ, JtR);
243  }
244 }
