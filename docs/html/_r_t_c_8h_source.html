362  void toTime(TIME &to, const TIMESTAMP &from) {
363  uint16_t year = 1970;
364  bool leapYear = false;
365  uint32_t seconds = from.timestamp + _tzOffset;
366 
367  to.timezone = _tzOffset;
368 
369  uint32_t secondsThisYear = 365 * 86400;
370  uint32_t secondsSoFar = 0;
371  while (seconds > secondsThisYear) {
372  seconds -= secondsThisYear;
373  secondsSoFar += secondsThisYear;
374 
375  if ((isLeapYear(++year))) {
376  secondsThisYear = 366 * 86400;
377  } else {
378  secondsThisYear = 365 * 86400;
379  }
380  }
381  to.year = year;
382 
383  uint8_t month;
384  leapYear = isLeapYear(year);
385  for (month = 0; month < sizeof(secondsFromYearStart); month++) {
386  if (leapYear) {
387  if (seconds < SECONDS_FROM_YEAR_START(month) + (1 == month) ? 86400 : 0) {
388  break;
389  }
390  } else {
391  if (seconds < SECONDS_FROM_YEAR_START(month)) {
392  break;
393  }
394  }
395  }
396  to.yearday = seconds / 86400;
397 
398  if (month) {
399  seconds -= SECONDS_FROM_YEAR_START(month - 1);
400  }
401 
402  if (leapYear && month > 1) {
403  seconds -= 86400;
404  }
405 
406  // Increment so we start numbering from 1
407  to.month = (Month)(month + 1);
408 
409  uint32_t day = seconds / 86400;
410  to.day = (uint8_t)day;
411  seconds -= day * 86400;
412  to.day += 1; // Increment so we start numbering from 1
413 
414  uint32_t hours = seconds / 3600;
415  to.hours = (uint8_t)hours;
416  seconds -= hours * 3600;
417 
418  uint32_t minutes = seconds / 60;

431  void toTimestamp(TIMESTAMP &to, const TIME &from) {
432  to.milliseconds = from.milliseconds;
433 
434  uint32_t seconds = from.seconds;
435 
436  seconds += (uint32_t)from.minutes * 60;
437 
438  seconds += (uint32_t)from.hours * 3600;
439 
440  if (from.yearday) {
441  seconds += from.yearday * 86400;
442  } else {
443  seconds += (from.day - 1) * 86400;
444  switch (from.month) {
445  case Month::JANUARY:
446  break;
447  case Month::FEBRUARY:
448  seconds += 5097600;
449  break;
450  default:
451  seconds += SECONDS_FROM_YEAR_START(from.month - 2);
452  if (isLeapYear(from.year)) {
453  seconds += 86400;
454  }
455  break;
456  }
457  }
458 
459  for (uint16_t year = from.year - 1; year >= 1970; year--) {
460  seconds += 365 * 86400;

462  seconds += 86400;
463  }
464  }

467  to.ticks = 0;

818  void handleEvents() {
819  uint8_t i;
820  TIMESTAMP timestamp;
821  current(timestamp);
822 
823  // Run any events pending
824  for (i = 0; i < _alarmCount && timestampGreaterThanOrEqual(timestamp, _alarms[i].when);
825  i++, current(timestamp)) {
826 // tx.printf(PSTR("Handing alarm at %d, current = %ld.%d, scheduled = %ld.%d\r\n"),
827 // i, timestamp.timestamp, timestamp.milliseconds,
828 // _alarms[i].when.timestamp, _alarms[i].when.milliseconds);
829 
830  _alarms[i].listener->alarm(AlarmSource::RTC);
831 
832  // Repeat the event if necessary
833  if (0 != _alarms[i].repeat.ticks || 0 != _alarms[i].repeat.milliseconds || 0 != _alarms[i].repeat.timestamp) {
834  timestampIncrement(_alarms[i].when, _alarms[i].repeat);
835  registerAlarm(_alarms[i]);
836  }
837  }
838 
839  // i is now the number of handled events
840  if (0 == i) {
841  return;
842  }
843 
844  // Shift remaining events down
845  if (i < _alarmCount) {
846  memcpy(_alarms, &(_alarms[i]), sizeof(*_alarms) * (uint8_t)(_alarmCount - i));
847  }

861  if (_alarmCount == alarmMax) {
862  return true;

864

866  uint8_t i;
867  for (i = 0; i < _alarmCount; ++i) {
868  if (timestampLessThan(alarm.when, _alarms[i].when)) {
869  break;

874  memmove(&_alarms[i + 1], &_alarms[i], FLAME_BYTESIZEOF(*_alarms) * (uint8_t)(_alarmCount - i));
875  memcpy(&_alarms[i], &alarm, FLAME_BYTESIZEOF(*_alarms));
876  } else {
877  memcpy(&(_alarms[_alarmCount]), &alarm, FLAME_BYTESIZEOF(*_alarms));
878  }
