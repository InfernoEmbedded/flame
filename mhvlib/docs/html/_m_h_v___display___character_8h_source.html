81  const char *p = string;
82 
83  while (*offsetX < 0) {

204  bool txAnimation(uint16_t row) {

276  void writeChar(char character) {
277  switch (character) {
278  case '\b':
279  if (_wrap && 0 == _currentCol) {
280  // Go to the previous line
281  if (_currentRow + 1 < rows) {
282  _currentRow++;
283  }
284  setCursor(cols - 1, _currentRow);
285  } else {
286  setCursor(_currentCol - 1, _currentRow);
287  }
288  break;
289  case '\t':
290  _currentCol = (_currentCol + 4) & ~(4 - 1);
291  if (_currentCol >= cols) {
292  if (_wrap) {
293  if (_currentRow > 0) {
294  _currentRow--;
295  } else if (_scroll) {
296  scrollVertically();
297  break;
298  }
299  _currentCol = 0;
300  } else {
301  _currentCol = cols - 1;
302  }
303  }
304  setCursor(_currentCol, _currentRow);
305  break;
306  case '\n':
307  if (_currentRow > 0) {
308  _currentRow--;
309  } else if (_scroll) {
310  scrollVertically();
311  }
312  _currentCol = 0;
313  setCursor(_currentCol, _currentRow);
314  break;
315  default:
316  if (_currentCol++ >= cols - 1 && _wrap) {
317  _currentCol = 0;
318  if (_currentRow > 0) {
319  _currentRow--;
320  } else if (_scroll) {
321  scrollVertically();
322  } else {
323  setCursor(_currentCol, _currentRow);
324  }
325  }
326  _writeChar(character);
327  break;
328  }
329  }

334  void scrollVertically() {
335  uint16_t x, y;
336  char c;
337 
338  for (y = rows - 2; y > 0; y--) {
339  // Copy each character in this row to the row above
340  for (x = 0; x < cols; ++x) {
341  _setCursor(x, y + 1);
342  c = _readChar();
343  _setCursor(x, y);
344  _writeChar(c);
345  }
346  }
347 
348  // Blank the last row
349  _setCursor(0, 0);
350  for (x = 0; x < cols; ++x) {
351  _writeChar(' ');
352  }
353 
354  // Set the cursor ready to write another line
355  setCursor(0, 0);
356  }
