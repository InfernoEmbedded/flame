63  void pushBits(uint8_t byte, bool rs) {
64  // setup RS
65  byte |= rs;
66 
67  // Load the byte into the shift register
68  _shifter.shiftOut(byte);
69 
70  // strobe Enable pin
71  _delay_loop_1(HD44780_TSU1);
72  mhv_pinOn(MHV_PIN_PARMS(enable));
73  _delay_loop_1(HD44780_TW);
74  mhv_pinOff(MHV_PIN_PARMS(enable));
75  _delay_loop_1(HD44780_TW);
76  }

83  void writeByte(uint8_t byte, bool rs) {
84  if (MHV_Display_HD44780<cols, rows, txBuffers>::_byteMode) {
85  // 8 bits interface mode
86  pushBits(byte, rs);
87  } else {
88  // 4 bits interface mode
89  // Write high order nibble
90  pushBits(byte & 0xF0, rs);
91  // Write low order nibble
92  pushBits(byte << 4, rs);
93  }
94  }
