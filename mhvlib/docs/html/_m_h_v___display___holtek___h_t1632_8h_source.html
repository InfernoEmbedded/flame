294  void flush() {
295  uint8_t x, y;
296  uint8_t bytes;
297  uint8_t *data;
298 
299  bytes = (MHV_HT1632_NMOS_32x8 == mode || MHV_HT1632_PMOS_32x8 == mode) ? 32 : 48;
300 
301  for (x = 0; x < arrayX; x++) {
302  for (y = 0; y < arrayY; y++) {
303  data = _frameBuffer + (uint16_t)bytes * (y * arrayY + x);
304  }
305  }
306 
307  for (x = 0; x < arrayX; x++) {
308  for (y = 0; y < arrayY; y++) {
309  data = _frameBuffer + (uint16_t)bytes * (y * arrayY + x);
310  outputStart(x, y);
311  _shifter.shiftOut(data, bytes);

324  void setPixel(uint16_t col, uint16_t row, uint8_t value) {
325  if (row < DISPLAY_Y && col < DISPLAY_X) {
326  // Coordinates of the display module
327  uint8_t moduleX, moduleY;
328 
329  moduleX = col / MODULE_X;
330  moduleY = row / MODULE_Y;
331 
332  // offsets within the module
333  col -= moduleX * MODULE_X;
334  row -= moduleY * MODULE_Y;
335 
336  uint8_t offset = col;
337  uint8_t bit = row;
338  switch (mode) {
339  case MHV_HT1632_NMOS_32x8:
340  case MHV_HT1632_PMOS_32x8:
341  break;
342  case MHV_HT1632_NMOS_24x16:
343  case MHV_HT1632_PMOS_24x16:
344  offset *= 2;
345  if (bit > 7) {
346  bit -= 8;
347  offset++;
348  }
349  break;
350  }
351 
352  // Change offset from the start of the display to the start of the framebuffer
353  offset += (moduleY * arrayX + moduleX) * DISPLAY_BYTES;
354 
355  if (value) {
356  _frameBuffer[offset] |= 1 << bit;
357  } else {
358  _frameBuffer[offset] &= ~(1 << bit);
359  }
360  }
361  }
