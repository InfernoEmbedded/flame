00032 /* Create a new H bridge where all transistors are controlled by us
00033  * Bottom transistors are toggled between VCC & ground.
00034  *
00035  * *******************************
00036  * UNTESTED - USE AT YOUR OWN RISK
00037  * *******************************
00038  *
00039  * A timer is required for magnitude control.
00040  *
00041  * If the H-bridge voltage is greater than VCC of the microcontroller:
00042  *   * resistors R1 & R2 and diodes D1 & D2 must be installed
00043  *   * Set type to MHV_SOFTWAREHBRIDGE_TYPE_PULLUP
00044  *   * Top transistors are toggled between high impedance & ground
00045  *
00046  * If the H bridge voltage is less than or equal to the microcontroller voltage,
00047  *  * resistors R1 & R2 can be omitted, and D1 and D2 can be shorted.
00048  *   * Set type to MHV_SOFTWAREHBRIDGE_TYPE_DIRECT
00049  *   * Top transistors are toggled between high impedance & ground
00050  *
00051  * The initial state is coasting
00052  */

00057                 MHV_DECLARE_PIN(dir2Bottom)) :
00058                 _type(type),
00059                 _timer(timer),

00074 /* Set up for a new timer pass
00075  * Care is taken in here to never allow the top and bottom of any side
00076  * to be on simultaneously
00077  */
00078 void MHV_SoftwareHBridge::reset() {
00079         switch (_direction) {
00080         case MHV_SOFTWAREHBRIDGE_DIR_COAST:
00081                 // Tops and bottoms are off
00082                 switch (_type) {
00083                 case MHV_SOFTWAREHBRIDGE_TYPE_PULLUP:
00084                         mhv_setInput(_dir1Top, _out1Top, NULL, _pin1Top, -1);
00085                         mhv_setInput(_dir2Top, _out2Top, NULL, _pin2Top, -1);
00086                         break;
00087                 case MHV_SOFTWAREHBRIDGE_TYPE_DIRECT:
00088                         mhv_pinOn(NULL, _out1Top, NULL, _pin1Top, -1);
00089                         mhv_pinOn(NULL, _out2Top, NULL, _pin2Top, -1);
00090                         break;
00091                 }
00092                 mhv_pinOff(NULL, _out1Bottom, NULL, _pin1Bottom, -1);
00093                 mhv_pinOff(NULL, _out2Bottom, NULL, _pin2Bottom, -1);
00094                 break;
00095         case MHV_SOFTWAREHBRIDGE_DIR_FORWARD:
00096                 // Top 1 is on, bottom 2 is on
00097                 mhv_pinOff(NULL, _out1Bottom, NULL, _pin1Bottom, -1);
00098                 switch (_type) {
00099                 case MHV_SOFTWAREHBRIDGE_TYPE_PULLUP:
00100                         mhv_setInput(_dir2Top, _out2Top, NULL, _pin2Top, -1);
00101                         mhv_setOutput(_dir1Top, _out1Top, NULL, _pin1Top, -1);
00102                         mhv_pinOff(_dir1Top, _out1Top, NULL, _pin1Top, -1);
00103                         break;
00104                 case MHV_SOFTWAREHBRIDGE_TYPE_DIRECT:
00105                         mhv_pinOn(NULL, _out2Top, NULL, _pin2Top, -1);
00106                         mhv_pinOff(NULL, _out1Top, NULL, _pin1Top, -1);
00107                         break;
00108                 }
00109                 mhv_pinOn(NULL, _out2Bottom, NULL, _pin2Bottom, -1);
00110                 break;
00111         case MHV_SOFTWAREHBRIDGE_DIR_BACKWARD:
00112                 // Top 2 is on, bottom 1 is on
00113                 mhv_pinOff(NULL, _out2Bottom, NULL, _pin2Bottom, -1);
00114                 switch (_type) {
00115                 case MHV_SOFTWAREHBRIDGE_TYPE_PULLUP:
00116                         mhv_setInput(_dir1Top, _out1Top, NULL, _pin1Top, -1);
00117                         mhv_setOutput(_dir2Top, _out2Top, NULL, _pin2Top, -1);
00118                         mhv_pinOff(_dir2Top, _out2Top, NULL, _pin2Top, -1);
00119                         break;
00120                 case MHV_SOFTWAREHBRIDGE_TYPE_DIRECT:
00121                         mhv_pinOn(NULL, _out1Top, NULL, _pin1Top, -1);
00122                         mhv_pinOff(NULL, _out2Top, NULL, _pin2Top, -1);
00123                         break;
00124                 }
00125                 mhv_pinOn(NULL, _out1Bottom, NULL, _pin1Bottom, -1);
00126                 break;
00127         case MHV_SOFTWAREHBRIDGE_DIR_BRAKE:
00128                 // Bottoms are on
00129                 switch (_type) {
00130                 case MHV_SOFTWAREHBRIDGE_TYPE_PULLUP:
00131                         mhv_setInput(_dir1Top, _out1Top, NULL, _pin1Top, -1);
00132                         mhv_setInput(_dir2Top, _out2Top, NULL, _pin2Top, -1);
00133                         break;
00134                 case MHV_SOFTWAREHBRIDGE_TYPE_DIRECT:
00135                         mhv_pinOn(NULL, _out1Top, NULL, _pin1Top, -1);
00136                         mhv_pinOn(NULL, _out2Top, NULL, _pin2Top, -1);
00137                         break;
00138                 }
00139                 mhv_pinOn(NULL, _out1Bottom, NULL, _pin1Bottom, -1);
00140                 mhv_pinOn(NULL, _out2Bottom, NULL, _pin2Bottom, -1);
00141                 break;
00142         }
00143 }

00145 /* Set the H bridge to coasting after the end of a timer cycle for PWM
00146  * This should be called from the trigger of the associated timer
00147  */
00148 void MHV_SoftwareHBridge::alarm() {
00149         // Set to coasting until the next reset
00150         switch (_type) {
00151         case MHV_SOFTWAREHBRIDGE_TYPE_PULLUP:
00152                 mhv_setInput(_dir1Top, _out1Top, NULL, _pin1Top, -1);
00153                 mhv_setInput(_dir2Top, _out2Top, NULL, _pin2Top, -1);
00154                 break;
00155         case MHV_SOFTWAREHBRIDGE_TYPE_DIRECT:
00156                 mhv_pinOn(NULL, _out1Top, NULL, _pin1Top, -1);
00157                 mhv_pinOn(NULL, _out2Top, NULL, _pin2Top, -1);
00158                 break;
00159         }
00160         mhv_pinOff(NULL, _out1Bottom, NULL, _pin1Bottom, -1);
00161         mhv_pinOff(NULL, _out2Bottom, NULL, _pin2Bottom, -1);
00162 }
