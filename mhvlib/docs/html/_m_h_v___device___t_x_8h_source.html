87  bool moreTX() {
88  // Call the completeFunction, if one is provided - allows the user to free memory, etc
89  if (NULL != _currentTx.completeFunction) {
90  _currentTx.completeFunction(_currentTx.data);
91  }
92 
93  if (_txPointers.consume(&_currentTx, sizeof(_currentTx))) {
94  _tx = NULL;
95  return false;
96  }
97 
98  _tx = _currentTx.data;
99 
100  return true;
101  }
102 
107  int nextCharacter() {
108  char c = '\0';
109 
110  if (NULL != _tx) {
111  if (_currentTx.progmem) {
112  c = pgm_read_byte(_tx);
113  } else {
114  c = *_tx;
115  }
116  }
117 
118  if ((_currentTx.isString && ('\0' == c)) ||
119  (!_currentTx.isString && (_tx == _currentTx.data + _currentTx.length))) {
120  if (!moreTX()) {
121  _tx = NULL;
122  _currentTx.data = NULL;
123  _currentTx.length = 0;
124  _currentTx.completeFunction = NULL;
125  _currentTx.progmem = false;
126  _currentTx.isString = false;
127  return -1;
128  }
129 
130  return nextCharacter();
131  }
132 
133  _tx++;
134  return (int)c;
135  }

324  bool debug(const char *file, int line, const char *function,
325  PGM_P format, ...) {
326  va_list ap;
327  va_start(ap, format);
328 
329  int length = snprintf_P(NULL, 0, PSTR("%s:%d\t%s():\t\t"),
330  file, line, function);
331  length += vsnprintf_P(NULL, 0, format, ap);
332  length += 3; // "\r\n\0"
333 
334  char *buf = (char *)malloc(length);
335  char *cur = buf;
336  if (NULL != cur) {
337  cur += snprintf_P(cur, length, PSTR("%s:%d\t%s():\t\t"),
338  file, line, function);
339  length -= cur - buf;
340  cur += vsnprintf_P(cur, length, format, ap);
341  *(cur++) = '\r';
342  *(cur++) = '\n';
343  *cur = '\0';
344 
345  if (write(buf, length, &mhv_device_tx_free)) {
346  mhv_device_tx_free(buf);
347  va_end(ap);
348  return true;
349  }
350 
351  va_end(ap);
352  return false;
353  }

355  va_end(ap);
356  return true;
