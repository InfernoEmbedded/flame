350  void toTime(MHV_TIME &to, const MHV_TIMESTAMP &from) {
351  uint16_t year = 1970;
352  bool leapYear = false;
353  uint32_t seconds = from.timestamp + _tzOffset;
354 
355  to.timezone = _tzOffset;
356 
357  uint32_t secondsThisYear = 365 * 86400;
358  uint32_t secondsSoFar = 0;
359  while (seconds > secondsThisYear) {
360  seconds -= secondsThisYear;
361  secondsSoFar += secondsThisYear;
362 
363  if ((mhv_isLeapYear(++year))) {
364  secondsThisYear = 366 * 86400;
365  } else {
366  secondsThisYear = 365 * 86400;
367  }
368  }
369  to.year = year;
370 
371  uint8_t month;
372  leapYear = mhv_isLeapYear(year);
373  for (month = 0; month < sizeof(mhv_secondsFromYearStart); month++) {
374  if (leapYear) {
375  if (seconds < pgm_read_dword(&mhv_secondsFromYearStart[month] + (1 == month) ? 86400 : 0)) {
376  break;
377  }
378  } else {
379  if (seconds < pgm_read_dword(&mhv_secondsFromYearStart[month])) {
380  break;
381  }
382  }
383  }
384  to.yearday = seconds / 86400;
385 
386  if (month) {
387  seconds -= pgm_read_dword(&mhv_secondsFromYearStart[month - 1]);
388  }
389 
390  if (leapYear && month > 1) {
391  seconds -= 86400;
392  }
393 
394  // Increment so we start numbering from 1
395  to.month = (MHV_MONTH)(month + 1);
396 
397  uint32_t day = seconds / 86400;
398  to.day = (uint8_t)day;
399  seconds -= day * 86400;
400  to.day += 1; // Increment so we start numbering from 1
401 
402  uint32_t hours = seconds / 3600;
403  to.hours = (uint8_t)hours;
404  seconds -= hours * 3600;
405 
406  uint32_t minutes = seconds / 60;
407  to.minutes = (uint8_t)minutes;
408  seconds -= minutes * 60;
409 
410  to.seconds = seconds;
411  to.milliseconds = from.milliseconds;
412  }

419  void toTimestamp(MHV_TIMESTAMP &to, const MHV_TIME &from) {
420  to.milliseconds = from.milliseconds;
421 
422  uint32_t seconds = from.seconds;
423 
424  seconds += (uint32_t)from.minutes * 60;
425 
426  seconds += (uint32_t)from.hours * 3600;
427 
428  if (from.yearday) {
429  seconds += from.yearday * 86400;
430  } else {
431  seconds += (from.day - 1) * 86400;
432  switch (from.month) {
433  case MHV_JANUARY:
434  break;
435  case MHV_FEBRUARY:
436  seconds += 5097600;
437  break;
438  default:
439  seconds += pgm_read_dword(&mhv_secondsFromYearStart[(uint8_t)from.month - 2]);
440  if (mhv_isLeapYear(from.year)) {
441  seconds += 86400;
442  }
443  break;
444  }
445  }
446 
447  for (uint16_t year = from.year - 1; year >= 1970; year--) {
448  seconds += 365 * 86400;
449  if (mhv_isLeapYear(year)) {
450  seconds += 86400;
451  }
452  }
453 
454  to.timestamp = seconds - _tzOffset;
455  to.ticks = 0;
456  }

463  void timestampIncrement(MHV_TIMESTAMP &timestamp, const MHV_TIMESTAMP &timestamp2) {
464  timestamp.ticks += timestamp2.ticks;
465  timestamp.milliseconds += timestamp2.milliseconds;
466 
467  while (timestamp.ticks > _ticksPerMillisecond) {

473  timestamp.timestamp++;
474  timestamp.milliseconds -= 1000;
475  }

477  timestamp.timestamp += timestamp2.timestamp;

796  void handleEvents() {
797  uint8_t i;
798  MHV_TIMESTAMP timestamp;
799  current(timestamp);
800 
801  // Run any events pending
802  for (i = 0; i < _alarmCount && mhv_timestampGreaterThanOrEqual(timestamp, _alarms[i].when);
803  i++, current(timestamp)) {
804  _alarms[i].listener->alarm();
805 
806  // Repeat the event if necessary
807  if (0 != _alarms[i].repeat.ticks || 0 != _alarms[i].repeat.milliseconds || 0 != _alarms[i].repeat.timestamp) {
808  current(_alarms[i].when);
809  timestampIncrement(_alarms[i].when, _alarms[i].repeat);
810  registerAlarm(_alarms[i]);
811  }
812  }
813 
814  if (0 == i) {
815  return;
816  }
817 
818  // Shift remaining events down
819  if (i < _alarmCount) {
820  mhv_memcpyTailFirst(_alarms, &(_alarms[i]), MHV_BYTESIZEOF(*_alarms), (uint8_t)(_alarmCount - i));
821  }
822 
823  _alarmCount -= i;
824  }

834  bool registerAlarm(MHV_ALARM &alarm) {
835  if (_alarmCount == alarmMax) {
836  return true;
837  }
838 
839  // Figure out where it needs to be inserted
840  uint8_t i;
841  for (i = 0; i < _alarmCount; ++i) {
842  if (mhv_timestampLessThan(alarm.when, _alarms[i].when)) {
843  break;
844  }
845  }
846  // i now is the offset of where the timestamp should be inserted
847  if (i < _alarmCount) {
848  mhv_memcpyTailFirst(_alarms + i + 1, _alarms + i, MHV_BYTESIZEOF(*_alarms), (uint8_t)(_alarmCount - i));
849  mhv_memcpy(_alarms + i, &alarm, MHV_BYTESIZEOF(*_alarms));
850  } else {
851  mhv_memcpy(&(_alarms[_alarmCount]), &alarm, MHV_BYTESIZEOF(*_alarms));
852  }
853 
854  _alarmCount++;
855 
856  return false;
857  }
