69  void writeCommand(MHV_HD44780_COMMAND command, uint8_t data) {
70  uint8_t byte = command | data;
71 
72  if (_mustDelay) {

140  uint8_t address;

198  void init(bool byteMode, bool multiLine, bool bigFont, bool cursorOn, bool cursorBlink,
199  bool left2right, bool scroll) {
200  /* Assume the worst case, we are called immediately after poweron
201  * Also assume that the MCU init takes 0 time
202  */
203  _delay_ms(HD44780_TINIT);
204 
205  // hardware initialization always set 8 bits mode
206  _byteMode = true;
207  uint8_t resetData = 1 << 4 | multiLine << 3 | bigFont << 2 | 1;
208  writeCommand(MHV_44780_CMD_SET_FUNCTION, resetData);
209  writeCommand(MHV_44780_CMD_SET_FUNCTION, resetData);
210  writeCommand(MHV_44780_CMD_SET_FUNCTION, resetData);
211 
212  _byteMode = byteMode;
213 
214 
215  function(byteMode, multiLine, bigFont);
216  _delay_us(39);
217  control(true, cursorOn, cursorBlink);
218  clear();
219  entryMode(left2right, scroll);
220  MHV_Display_Character<cols, rows, txBuffers>::_currentRow = rows - 1;
221  MHV_Display_Character<cols, rows, txBuffers>::_currentCol = 0;
222  }
