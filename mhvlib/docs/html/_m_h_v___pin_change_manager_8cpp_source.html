68 void MHV_PinChangeManager::pinChange(uint8_t offset) {
69  MHV_EVENT_PIN *pin = _pins + offset;
70  MHV_EVENT_PIN *maxPin = pin + 8;
71 
72  for (; pin < maxPin; ++pin) {
73  if (NULL == pin->listener) {
74  continue;
75  }
76 
77  bool cur = _MMIO_BYTE(pin->port) & pin->mask;
78  if (cur != pin->previous) {
79  // Pin has changed
80  pin->previous = cur;
81  pin->changed = 1;
82  pin->listener->pinChanged(pin - _pins, cur);
83  }
84  }
85 }
