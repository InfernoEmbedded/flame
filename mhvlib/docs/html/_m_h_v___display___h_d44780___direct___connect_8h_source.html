108  void writeNibble(uint8_t nibble, bool rs) {
109  uint8_t oldPort = _SFR_IO8(dataOut) & ~(_dataMask);
110  _SFR_IO8(controlOut) &= ~(HD44780_RW);
111  if (rs) {
112  _SFR_IO8(controlOut) |= HD44780_RS;
113  }
114 
115  _SFR_IO8(dataOut) = oldPort | ((nibble << dataPin) & _dataMask);
116 
117  _delay_loop_1(HD44780_TSU1);
118  _SFR_IO8(controlOut) |= HD44780_E;
119  _delay_loop_1(HD44780_TW);
120  _SFR_IO8(controlOut) &= ~HD44780_E & ~HD44780_RS;
121  _delay_loop_1(HD44780_TW);
