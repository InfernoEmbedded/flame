89 uint16_t MHV_ADC::busyRead(int8_t channel, uint8_t reference) {
90  ADMUX = reference | (channel & 0x0F);
91 
92 #ifdef MUX5
93  ADCSRB = (ADCSRB & ~_BV(MUX5)) | ((channel & _BV(5) >> (5-MUX5)));
94 #endif
95 
96 // Start the conversion
97  ADCSRA |= _BV(ADSC);
98 
99  while (ADCSRA & _BV(ADSC)) {};
100 
101  return ADC;
102 }
