00054 /* Calibrate the RC oscillator to 8.25 MHz. The core clock of 16.5 MHz is
00055  * derived from the 66 MHz peripheral clock by dividing. Our timing reference
00056  * is the Start Of Frame signal (a single SE0 bit) available immediately after
00057  * a USB RESET. We first do a binary search for the OSCCAL value and then
00058  * optimize this value with a neighboorhod search.
00059  * This algorithm may also be used to calibrate the RC oscillator directly to
00060  * 12 MHz (no PLL involved, can therefore be used on almost ALL AVRs), but this
00061  * is wide outside the spec for the OSCCAL value and the required precision for
00062  * the 12 MHz clock! Use the RC oscillator calibrated to 12 MHz for
00063  * experimental purposes only!
00064  */
00065 static void calibrateOscillator(void)
00066 {
00067 uchar       step = 128;
00068 uchar       trialValue = 0, optimumValue;
00069 int         x, optimumDev, targetValue = (unsigned)(1499 * (double)F_CPU / 10.5e6 + 0.5);
00070 
00071     /* do a binary search: */
00072     do{
00073         OSCCAL = trialValue + step;
00074         x = usbMeasureFrameLength();    /* proportional to current real frequency */
00075         if(x < targetValue)             /* frequency still too low */
00076             trialValue += step;
00077         step >>= 1;
00078     }while(step > 0);
00079     /* We have a precision of +/- 1 for optimum OSCCAL here */
00080     /* now do a neighborhood search for optimum value */
00081     optimumValue = trialValue;
00082     optimumDev = x; /* this is certainly far away from optimum */
00083     for(OSCCAL = trialValue - 1; OSCCAL <= trialValue + 1; OSCCAL++){
00084         x = usbMeasureFrameLength() - targetValue;
00085         if(x < 0)
00086             x = -x;
00087         if(x < optimumDev){
00088             optimumDev = x;
00089             optimumValue = OSCCAL;
00090         }
00091     }
00092     OSCCAL = optimumValue;
00093 }
00094 /*
00095 Note: This calibration algorithm may try OSCCAL values of up to 192 even if
00096 the optimum value is far below 192. It may therefore exceed the allowed clock
00097 frequency of the CPU in low voltage designs!
00098 You may replace this search algorithm with any other algorithm you like if
00099 you have additional constraints such as a maximum CPU clock.
00100 For version 5.x RC oscillators (those with a split range of 2x128 steps, e.g.
00101 ATTiny25, ATTiny45, ATTiny85), it may be useful to search for the optimum in
00102 both regions.
00103 */
