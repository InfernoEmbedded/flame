52 /* Calibrate the RC oscillator to 8.25 MHz. The core clock of 16.5 MHz is
53  * derived from the 66 MHz peripheral clock by dividing. Our timing reference
54  * is the Start Of Frame signal (a single SE0 bit) available immediately after
55  * a USB RESET. We first do a binary search for the OSCCAL value and then
56  * optimize this value with a neighboorhod search.
57  * This algorithm may also be used to calibrate the RC oscillator directly to
58  * 12 MHz (no PLL involved, can therefore be used on almost ALL AVRs), but this
59  * is wide outside the spec for the OSCCAL value and the required precision for
60  * the 12 MHz clock! Use the RC oscillator calibrated to 12 MHz for
61  * experimental purposes only!
62  */
63 static void calibrateOscillator(void)
64 {
65 uchar step = 128;
66 uchar trialValue = 0, optimumValue;
67 int x, optimumDev, targetValue = (unsigned)(1499 * (double)F_CPU / 10.5e6 + 0.5);
68 
69  /* do a binary search: */
70  do{
71  OSCCAL = trialValue + step;
72  x = usbMeasureFrameLength(); /* proportional to current real frequency */
73  if(x < targetValue) /* frequency still too low */
74  trialValue += step;
75  step >>= 1;
76  }while(step > 0);
77  /* We have a precision of +/- 1 for optimum OSCCAL here */
78  /* now do a neighborhood search for optimum value */
79  optimumValue = trialValue;
80  optimumDev = x; /* this is certainly far away from optimum */
81  for(OSCCAL = trialValue - 1; OSCCAL <= trialValue + 1; OSCCAL++){
82  x = usbMeasureFrameLength() - targetValue;
83  if(x < 0)
84  x = -x;
85  if(x < optimumDev){
86  optimumDev = x;
87  optimumValue = OSCCAL;
88  }
89  }
90  OSCCAL = optimumValue;
91 }
92 /*
93 Note: This calibration algorithm may try OSCCAL values of up to 192 even if
94 the optimum value is far below 192. It may therefore exceed the allowed clock
95 frequency of the CPU in low voltage designs!
96 You may replace this search algorithm with any other algorithm you like if
97 you have additional constraints such as a maximum CPU clock.
98 For version 5.x RC oscillators (those with a split range of 2x128 steps, e.g.
99 ATTiny25, ATTiny45, ATTiny85), it may be useful to search for the optimum in
100 both regions.
101 */

103 void usbEventResetReady(void)

113 static unsigned char mhv_vusbIdleRate = 1; /* in 4 ms units */
114 
115 unsigned char usbFunctionSetup(uchar data[8]) {
116  usbRequest_t *rq = (usbRequest_t *) ((void *) data);
117 
118  usbMsgPtr = NULL;
119  if ((rq->bmRequestType & USBRQ_TYPE_MASK) == USBRQ_TYPE_CLASS) { /* class request type */
120  if (rq->bRequest == USBRQ_HID_GET_REPORT) { /* wValue: ReportType (highbyte), ReportID (lowbyte) */
121  return 0;
122  } else if (rq->bRequest == USBRQ_HID_GET_IDLE) {
123  usbMsgPtr = &mhv_vusbIdleRate;
124  return 1;
125  } else if (rq->bRequest == USBRQ_HID_SET_IDLE) {
126  mhv_vusbIdleRate = rq->wValue.bytes[1];
127  }
128  } else {
129  /* no vendor specific requests implemented */
130  }
131  return 0;
132 }
