00077 void MHV_Display_HD44780_Shift_Register::pushBits(uint8_t byte, bool rs) {
00078 // setup RS
00079         byte |= (rs ? 0x01 : 0x00);
00080 
00081 // Load the byte into the shift register
00082         MHV_SHIFTOUT_BYTE(byte);
00083 
00084 // strobe Enable pin
00085         _delay_loop_1(HD44780_TSU1);
00086         mhv_pinOn(0, _enableOut, 0, _enablePin, -1);
00087         _delay_loop_1(HD44780_TW);
00088         mhv_pinOff(0, _enableOut, 0, _enablePin, -1);
00089         _delay_loop_1(HD44780_TW);
