294  break;
295  case MHV_TIMER_8_PWM_PHASE_CORRECT_VAR_FREQ:

297  _SFR_MEM8(controlRegB) = (_SFR_MEM8(controlRegB) & 0xf7) | _BV(WGM02);
298  break;
299  case MHV_TIMER_8_PWM_PHASE_CORRECT_2_OUTPUT:
300  _SFR_MEM8(controlRegA) = (_SFR_MEM8(controlRegA) & 0xfc) | _BV(WGM01);
301  _SFR_MEM8(controlRegB) = (_SFR_MEM8(controlRegB) & 0xf7);
302  break;
303  case MHV_TIMER_8_PWM_FAST_VAR_FREQ:
304  _SFR_MEM8(controlRegA) = (_SFR_MEM8(controlRegA) & 0xfc) | _BV(WGM01) | _BV(WGM00);
305  _SFR_MEM8(controlRegB) = (_SFR_MEM8(controlRegB) & 0xf7) | _BV(WGM02);
306  break;
307  case MHV_TIMER_8_PWM_FAST_2_OUTPUT:

372  uint8_t calculatePrescaler(uint32_t time, MHV_TIMER_PRESCALER *prescaler, uint16_t *factor) {
373  uint32_t limit = 0;
374  if (8 == bits) {
375  limit = 256L;
376  } else if (16 == bits) {
377  limit = 65536L;
378  }
379 
380  switch (type) {
381  case MHV_TIMER_5_PRESCALERS:
382  if (time <= limit) {
383  *prescaler = MHV_TIMER_PRESCALER_5_1;
384  *factor = 1;
385  } else if (time <= limit * 8) {
386  *prescaler = MHV_TIMER_PRESCALER_5_8;
387  *factor = 8;
388  } else if (time <= limit * 64) {
389  *prescaler = MHV_TIMER_PRESCALER_5_64;
390  *factor = 64;
391  } else if (time <= limit * 256) {
392  *prescaler = MHV_TIMER_PRESCALER_5_256;
393  *factor = 256;
394  } else if (time <= limit * 1024) {
395  *prescaler = MHV_TIMER_PRESCALER_5_1024;
396  *factor = 1024;
397  } else {
398  return 1;
399  }
400  break;
401 
402  case MHV_TIMER_7_PRESCALERS:
403  if (time <= limit) {
404  *prescaler = MHV_TIMER_PRESCALER_7_1;
405  *factor = 1;
406  } else if (time <= limit * 8) {
407  *prescaler = MHV_TIMER_PRESCALER_7_8;
408  *factor = 8;
409  } else if (time <= limit * 32) {
410  *prescaler = MHV_TIMER_PRESCALER_7_32;
411  *factor = 32;
412  } else if (time <= limit * 64) {
413  *prescaler = MHV_TIMER_PRESCALER_7_64;
414  *factor = 64;
415  } else if (time <= limit * 128) {
416  *prescaler = MHV_TIMER_PRESCALER_7_128;
417  *factor = 128;
418  } else if (time <= limit * 256) {
419  *prescaler = MHV_TIMER_PRESCALER_7_256;
420  *factor = 256;
421  } else if (time <= limit * 1024) {
422  *prescaler = MHV_TIMER_PRESCALER_7_1024;
423  *factor = 1024;
424  } else {
425  return 1;
426  }
427  break;
428  }
429 
430  return 0;
431  }
