68 void MHV_Debounce::checkHeld() {
69  MHV_DEBOUNCE_PIN *pin = _pins;
70  MHV_DEBOUNCE_PIN *maxPin = pin + MHV_PC_INT_COUNT;
71 
72  for (; pin < maxPin; ++pin) {
73  if (pin->timestamp.timestamp && !pin->previous) {
74 // Pin is currently held down
75  MHV_TIMESTAMP heldFor;
76  _rtc.elapsed(pin->timestamp, heldFor);
77 
78  if (!pin->held && mhv_timestampGreaterThanOrEqual(heldFor, _heldTime)) {
79  pin->held = true;
80  pin->listener->heldDown(pin - _pins, &heldFor);
81  _rtc.current(pin->timestamp);
82  } else if (pin->held && mhv_timestampGreaterThanOrEqual(heldFor, _repeatTime)) {
83  pin->listener->heldDown(pin - _pins, &heldFor);
84  _rtc.current(pin->timestamp);
85  }
86  }
87  }
88 }

95 void MHV_Debounce::pinChanged(uint8_t pcInt, bool newState) {
96  if (newState) {
97 // Pin has been released
98  MHV_TIMESTAMP heldFor;
99 
100 /* Call the singlePress function if the time is greater than the debounce time,
101  * and the button has not been held down
102  */
103  _rtc.elapsed(_pins[pcInt].timestamp, heldFor);
104  if (!_pins[pcInt].held && mhv_timestampGreaterThanOrEqual(heldFor, _debounceTime)) {
105  _pins[pcInt].listener->singlePress(pcInt, &heldFor);
106  }
107 
108  _pins[pcInt].held = false;
109  _pins[pcInt].timestamp.timestamp = 0;
110  } else {
111 // Pin has been pressed
112  _rtc.current(_pins[pcInt].timestamp);
113  }
114 }
