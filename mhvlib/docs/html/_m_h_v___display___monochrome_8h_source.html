64  bool writeChar(const MHV_FONT &font, int16_t *offsetX, int16_t offsetY,
65  uint8_t onValue, uint8_t offValue, char character) {
66  uint8_t c = (uint8_t)character;
67  if (c < font.firstChar || c >= font.firstChar + font.charCount) {
68  c = font.unknown;
69  }
70 
71  // Normalize character to font
72  c -= font.firstChar;
73  uint8_t charWidth = pgm_read_byte(font.widths + c);
74 
75  const uint8_t *fontChar = font.fontData + pgm_read_word(font.offsets + c);
76 
77  // Render each column
78  bool ret = false;
79  do {
80  int8_t bit = 7;
81  uint8_t y = 0;
82  uint8_t data = pgm_read_byte(fontChar++);
83 
84  if (*offsetX >= 0) {
85  // Start at the bottom of the column and work up
86  for (; y < font.maxHeight; y++, bit--) {
87  ret = true;
88 
89  if (data & (1 << bit)) {
90  setPixel(*offsetX, offsetY + y, onValue);
91  } else {
92  setPixel(*offsetX, offsetY + y, offValue);
93  }
94 
95  // Support multiple-byte font images - fetch the next byte of data if we exhaust the current one
96  if (-1 == bit && (font.maxHeight - y) > 0) {
97  bit = 7;
98  data = pgm_read_byte(fontChar++);
99  }
100  }
101  }
102  } while (++(*offsetX) < (int16_t)cols && --charWidth);
103 
104  *offsetX += charWidth;
105 
106  return ret;
107  }

117  bool writeSeperator(const MHV_FONT &font, int16_t *offsetX, int16_t offsetY,
118  uint8_t offValue) {
119  bool ret = false;
120 
121  if (*offsetX >= 0 && *offsetX < (int16_t)cols) {
122  // Start at the bottom of the column and work up
123  uint8_t y = 0;
124  uint8_t row;
125 
126  while (y++ < font.maxHeight) {
127  row = offsetY + y;
128  if (row < rows) {
129  setPixel(*offsetX, row, offValue);
130  ret = true;
131  }
132  }
133  }
134 
135  (*offsetX)++;
136  return ret;
137  }

291  bool txAnimation(const MHV_FONT &font, int16_t offsetY, uint8_t onValue, uint8_t offValue) {
292  int16_t offsetX = _txOffset--;
293 
294  clear(offValue);
295 
296  bool ret;
297  if (MHV_Device_TX::_currentTx.progmem) {
298  if (MHV_Device_TX::_currentTx.isString) {
299  ret = writeString_P(font, &offsetX, offsetY, onValue, offValue, MHV_Device_TX::_tx);
300  } else {
301  ret = writeBuffer_P(font, &offsetX, offsetY, onValue, offValue, MHV_Device_TX::_tx, MHV_Device_TX::_currentTx.length);
302  }
303  } else {
304  if (MHV_Device_TX::_currentTx.isString) {
305  ret = writeString(font, &offsetX, offsetY, onValue, offValue, MHV_Device_TX::_tx);
306  } else {
307  ret = writeBuffer(font, &offsetX, offsetY, onValue, offValue, MHV_Device_TX::_tx, MHV_Device_TX::_currentTx.length);
308  }
309  }
310 
311  if (!ret) {
312  _txOffset = cols - 1;
313  return MHV_Device_TX::moreTX();
314  }
315 
316  return true;
317  }
