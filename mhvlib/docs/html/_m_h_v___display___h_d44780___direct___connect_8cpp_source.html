00163 void MHV_Display_HD44780_Direct_Connect::writeNibble(uint8_t nibble, bool rs) {
00164         uint8_t oldPort = *_dataOut & ~(_dataMask);
00165         *_controlOut &= ~(HD44780_RW);
00166         if (rs) {
