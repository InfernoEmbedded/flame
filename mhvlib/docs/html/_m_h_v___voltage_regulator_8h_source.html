79  /* Regulate as a boosting regulator
80  */
81  inline void regulateBoost() {
82  uint16_t adc = mhv_ad_busyRead(adcChannel, vref);
83  uint16_t newPWM = (uint16_t)(_pwm * (float)_targetADC / (float)adc);
84 
85  if (newPWM == _pwm) {
86  if (_targetADC < adc) {
87  newPWM = _pwm - 1;
88  } else if (_targetADC > adc) {
89  newPWM = _pwm + 1;
90  }
91  }
92 
93  // set a minimum on and off time - boost regulators are useless if the transistor is always on or off
94  if (newPWM < 1) {
95  newPWM = 1;
96  } else if (newPWM + 1 > _timer.getTop()) {
97  newPWM = _timer.getTop() - 1;
98  }
99 
100  _pwm = newPWM;
101  _timer.setOutput2(_pwm);
102 
103  _lastADC = adc;
104  }
105 
109  inline void regulateBuck() {
110  uint16_t adc = mhv_ad_busyRead(adcChannel, vref);
111 
112  if (adc < _targetADC) {
113  _lastMoveUp = true;
114  } else {
115  _lastMoveUp = false;
116  }
117 
118  // _lastMoveUp now contains the move we want to make
119  uint16_t newPWM;
120  if (adc != _targetADC) {
121  if ((!_invert && !_lastMoveUp) || (_invert && _lastMoveUp)) {
122  newPWM = _pwm - 1;
123  } else {
124  newPWM = _pwm + 1;
125  }
126 
127  // set a minimum on and off time
128  if (newPWM < 1) {
129  newPWM = 1;
130  } else if (newPWM > uint16_t(_timer.getTop() - 1)) {
131  newPWM = _timer.getTop() - 1;
132  }
133 
134  _pwm = newPWM;
135  _timer.setOutput2(_pwm);
136  }
137 
138  _lastADC = adc;
139  }

145  inline void regulate() {
146  switch (mode) {
147  case MHV_VREG_MODE_BOOST:
148  regulateBoost();
149  break;
150  case MHV_VREG_MODE_BUCK:
151  regulateBuck();
152  break;
153  }
154  }
