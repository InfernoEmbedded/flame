00061         switch (voice.phase) {
00062         case MHV_VOICE_ATTACK:
00063                 if (voice.phaseOffset == voice.instrument->attackTime) {
00064                         voice.phaseOffset = 0;

00067                 break;
00068         case MHV_VOICE_DECAY1:
00069                 if (voice.phaseOffset == voice.instrument->decay1Time) {
00070                         voice.phaseOffset = 0;

00073                 break;

00109 
00110         switch (voice.phase) {

00112                 amplitude *= voice.phaseOffset;
00113                 amplitude /= voice.instrument->attackTime;
00114                 voice.amplitude = amplitude;

00117                 amplitude -= voice.phaseOffset * (voice.velocity - voice.instrument->decay1Amount) / voice.instrument->decay1Time;

00119                 break;
00120         case MHV_VOICE_SUSTAIN1:
00121                 amplitude *= voice.instrument->decay1Amount;

00147 inline MHV_SAMPLE MHV_WaveGenerator::renderVoice(MHV_VOICE &voice) {
00148         if (voice.phase == MHV_VOICE_AVAILABLE) {
00149                 return 0;
00150         }
00151 
00152         uint8_t sampleOffset = uint32_t(voice.currentOffset * voice.frequency / 440)  % 72;
00153         MHV_SAMPLE sample = pgm_read_byte(voice.instrument->samples + sampleOffset);
00154 
00155 // Calculate amplitude
00156         MHV_AMPLITUDE amplitude = currentAmplitude(voice);
